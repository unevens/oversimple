374  int processBlock(float* const* input,
375  int numChannels,
376  int numSamples,
377  ScalarBuffer<float>& output)
378  {
379  floatToDoubleBuffer.setNumChannelsAndSamples(numChannels, numSamples);
380  doubleToFloatBuffer.setNumChannelsAndSamples(
381  numChannels, std::ceil(numSamples * oversamplingRate));
382  for (int c = 0; c < numChannels; ++c) {
383  for (int i = 0; i < numSamples; ++i) {
384  floatToDoubleBuffer[c][i] = (double)input[c][i];
385  }
386  }
387  int samples = FirUnbufferedResampler::processBlock(floatToDoubleBuffer,
388  doubleToFloatBuffer);
389  copyScalarBuffer(doubleToFloatBuffer, output);
390  return samples;
391  }
392

400  ScalarBuffer<float>& output)
401  {
402  return processBlock(
