371 
372 public:
383  explicit TUpSampler(uint32_t numChannels,
384  double transitionBand = 4.0,
385  uint32_t fftSamplesPerBlock = 256,
386  double oversamplingRate = 1.0)
387  : UpSampler(numChannels, transitionBand, fftSamplesPerBlock, oversamplingRate)

399  uint32_t processBlock(float* const* input, uint32_t numSamples)
400  {
401  auto const numUpSampledSamples = (uint32_t)std::ceil(numSamples * oversamplingRate);
402  assert(floatToDoubleBuffer.getCapacity() >= numSamples);
403  assert(doubleToFloatBuffer.getCapacity() >= numUpSampledSamples);
404  floatToDoubleBuffer.setNumSamples(numSamples);
405  doubleToFloatBuffer.setNumSamples(numUpSampledSamples);
406  for (uint32_t c = 0; c < numChannels; ++c) {
407  for (uint32_t i = 0; i < numSamples; ++i) {
408  floatToDoubleBuffer[c][i] = (double)input[c][i];
409  }
410  }
411  auto const samples = UpSampler::processBlock(floatToDoubleBuffer.get(), numSamples);
412  copyBuffer(output, doubleToFloatBuffer);
413  return samples;
414  }

515  void processBlock(float* const* input, uint32_t numSamples, float** output, uint32_t requiredSamples)
516  {
517  assert(floatToDoubleBuffer.getCapacity() >= numSamples);
518  assert(doubleToFloatBuffer.getCapacity() >= requiredSamples);
519  floatToDoubleBuffer.setNumSamples(numSamples);
520  doubleToFloatBuffer.setNumSamples(requiredSamples);
521  if (numSamples > 0) {
522  for (uint32_t c = 0; c < numChannels; ++c) {
523  std::copy(&input[c][0], &input[c][0] + numSamples, &floatToDoubleBuffer[c][0]);
524  }
525  }
526  DownSampler::processBlock(floatToDoubleBuffer.get(), numSamples, doubleToFloatBuffer.get(), requiredSamples);
527  for (uint32_t c = 0; c < numChannels; ++c) {
528  for (uint32_t i = 0; i < requiredSamples; ++i) {
529  output[c][i] = (float)doubleToFloatBuffer[c][i];
530  }
531  }
532  }

546  DownSampler::processBlock(
547  floatToDoubleBuffer.get(), floatToDoubleBuffer.getNumSamples(), doubleToFloatBuffer.get(), requiredSamples);
