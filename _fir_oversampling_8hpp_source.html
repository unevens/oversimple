361  int ProcessBlock(float* const* input,
362  int numChannels,
363  int numSamples,
364  ScalarBuffer<float>& output)
365  {
366  floatToDoubleBuffer.SetNumChannelsAndSize(numChannels, numSamples);
367  doubleToFloatBuffer.SetNumChannelsAndSize(
368  numChannels, std::ceil(numSamples * oversamplingFactor));
369  for (int c = 0; c < numChannels; ++c) {
370  for (int i = 0; i < numSamples; ++i) {
371  floatToDoubleBuffer[c][i] = (double)input[c][i];
372  }
373  }
374  int samples = FirUnbufferedResampler::ProcessBlock(floatToDoubleBuffer,
375  doubleToFloatBuffer);
376  CopyScalarBuffer(doubleToFloatBuffer, output);
377  return samples;
378  }
379 
386  int ProcessBlock(ScalarBuffer<float> const& input,
387  ScalarBuffer<float>& output)
388  {
389  return ProcessBlock(
390  input.Get(), input.GetNumChannels(), input.GetSize(), output);
391  }
