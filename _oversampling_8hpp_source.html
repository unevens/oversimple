300  uint32_t upSample(Float* const* input, uint32_t numSamples)
301  {
302  assert(settings.upSampleInputBufferType == BufferType::plain);
303  if (settings.isUsingLinearPhase) {
304  auto const numUpSampledSamples = firUpSampler.processBlock(input, numSamples);
305  if (settings.upSampleOutputBufferType == BufferType::interleaved) {
306  assert(firUpSampler.getOutput().getNumSamples() == numUpSampledSamples);
307  assert(upSampleOutputInterleaved.getCapacity() >= numUpSampledSamples);
308  upSampleOutputInterleaved.setNumSamples(numUpSampledSamples);
309  bool const ok = upSampleOutputInterleaved.interleave(firUpSampler.getOutput());
310  assert(ok);
311  }
312  return numUpSampledSamples;
313  }
314  else {
315  iirUpSampler.processBlock(input, numSamples);
316  auto const numUpSampledSamples = numSamples * getOversamplingRate();
317  if (settings.upSampleOutputBufferType == BufferType::plain) {
318  assert(numUpSampledSamples == iirUpSampler.getOutput().getNumSamples());
319  assert(upSamplePlainBuffer.getCapacity() >= numUpSampledSamples);
320  upSamplePlainBuffer.setNumSamples(numUpSampledSamples);
321  iirUpSampler.getOutput().deinterleave(upSamplePlainBuffer);
322  }
323  return numUpSampledSamples;
324  }

415  void downSample(Float* const* input, uint32_t numInputSamples, Float** output, uint32_t numOutputSamples)
416  {
417  assert(settings.downSampleOutputBufferType == BufferType::plain);
418  assert(settings.downSampleInputBufferType == BufferType::plain);
419  if (settings.isUsingLinearPhase) {
420  firDownSampler.processBlock(input, numInputSamples, output, numOutputSamples);
421  }
422  else {
423  assert(numOutputSamples * (1 << settings.order) == numInputSamples);
424  assert(downSampleBufferInterleaved.getCapacity() >= numInputSamples);
425  downSampleBufferInterleaved.setNumSamples(numInputSamples);
426  bool const ok = downSampleBufferInterleaved.interleave(input, settings.numDownSampledChannels, numInputSamples);
427  assert(ok);
428  iirDownSampler.processBlock(downSampleBufferInterleaved);
429  iirDownSampler.getOutput().deinterleave(output, settings.numDownSampledChannels, numOutputSamples);
430  }
431  }

440  {

443  if (settings.isUsingLinearPhase) {
444  auto const numInputSamples = input.getNumSamples();

446  downSamplePlainInputBuffer.setNumSamples(numInputSamples);
447  input.deinterleave(downSamplePlainInputBuffer);
448  firDownSampler.processBlock(downSamplePlainInputBuffer, output, numOutputSamples);
449  }

464  void downSample(Float* const* input, uint32_t numInputSamples, uint32_t numOutputSamples)
465  {
466  assert(settings.downSampleOutputBufferType == BufferType::interleaved);
467  assert(settings.downSampleInputBufferType == BufferType::plain);
468  if (settings.isUsingLinearPhase) {
469  assert(downSamplePlainOutputBuffer.getCapacity() >= numOutputSamples);
470  assert(downSampleBufferInterleaved.getCapacity() >= numOutputSamples);
471  downSamplePlainOutputBuffer.setNumSamples(numOutputSamples);
472  downSampleBufferInterleaved.setNumSamples(numOutputSamples);
473  firDownSampler.processBlock(input, numInputSamples, downSamplePlainOutputBuffer.get(), numOutputSamples);
474  downSampleBufferInterleaved.setNumSamples(numOutputSamples);
475  bool const ok = downSampleBufferInterleaved.interleave(downSamplePlainOutputBuffer);
476  assert(ok);
477  }
478  else {
479  assert(numOutputSamples * (1 << settings.order) == numInputSamples);
480  assert(downSampleBufferInterleaved.getCapacity() >= numOutputSamples);
481  downSampleBufferInterleaved.setNumSamples(numInputSamples);
482  bool const ok = downSampleBufferInterleaved.interleave(input, settings.numDownSampledChannels, numInputSamples);
483  assert(ok);

562  void prepareInternalBuffers()
563  {
564  iirUpSampler.prepareBuffers(settings.maxNumInputSamples);
565  iirDownSampler.prepareBuffers(settings.maxNumInputSamples);
566  firUpSampler.prepareBuffers(settings.maxNumInputSamples);
567  auto const maxFirUpSampledSamples = firUpSampler.getMaxNumOutputSamples();
568  firDownSampler.prepareBuffers(maxFirUpSampledSamples, settings.maxNumInputSamples);
569  auto const maxSamplesUpSampled = settings.maxNumInputSamples * (1 << settings.maxOrder);
570  auto const maxSamples = std::max(maxFirUpSampledSamples, maxSamplesUpSampled);
571  downSampleBufferInterleaved.reserve(maxSamples);
572  downSamplePlainOutputBuffer.reserve(maxSamples);
573  downSamplePlainInputBuffer.reserve(maxSamples);
574  upSampleOutputInterleaved.reserve(maxSamples);
575  upSamplePlainBuffer.reserve(maxSamples);

578  void setupInputOutputBuffers()
579  {
580  if (settings.upSampleOutputBufferType == BufferType::interleaved) {
581  upSampleOutputInterleaved.setNumChannels(settings.numUpSampledChannels);
582  }
583  else {
584  upSampleOutputInterleaved.setNumChannels(0);
585  }
586  if (settings.upSampleOutputBufferType == BufferType::plain ||
587  settings.upSampleInputBufferType == BufferType::interleaved)
588  {
589  upSamplePlainBuffer.setNumChannels(settings.numUpSampledChannels);
590  }
591  else {
592  upSamplePlainBuffer.setNumChannels(0);
593  }
594  if (settings.downSampleOutputBufferType == BufferType::plain &&
595  settings.downSampleInputBufferType == BufferType::plain) {
596  downSampleBufferInterleaved.setNumChannels(settings.numDownSampledChannels);
597  downSamplePlainInputBuffer.setNumChannels(0);
598  downSamplePlainOutputBuffer.setNumChannels(0);
599  }
600  else if (settings.downSampleOutputBufferType == BufferType::interleaved &&
601  settings.downSampleInputBufferType == BufferType::interleaved)
602  {
603  downSampleBufferInterleaved.setNumChannels(settings.numDownSampledChannels);
604  downSamplePlainInputBuffer.setNumChannels(settings.numDownSampledChannels);
605  downSamplePlainOutputBuffer.setNumChannels(settings.numDownSampledChannels);
606  }
607  else if (settings.downSampleOutputBufferType == BufferType::plain &&
608  settings.downSampleInputBufferType == BufferType::interleaved)
609  {
610  downSampleBufferInterleaved.setNumChannels(0);
611  downSamplePlainInputBuffer.setNumChannels(settings.numDownSampledChannels);
612  downSamplePlainOutputBuffer.setNumChannels(settings.numDownSampledChannels);
613  }
614  else if (settings.downSampleOutputBufferType == BufferType::interleaved &&
615  settings.downSampleInputBufferType == BufferType::plain)
616  {
617  downSampleBufferInterleaved.setNumChannels(settings.numDownSampledChannels);
618  downSamplePlainInputBuffer.setNumChannels(0);
619  downSamplePlainOutputBuffer.setNumChannels(settings.numDownSampledChannels);
