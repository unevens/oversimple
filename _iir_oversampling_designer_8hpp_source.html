128 inline std::string
129 IirOversamplingDesigner::print() const
130 {
131  std::string text;
132  int i = 0;
133  for (auto& stage : stages) {
134  text += "stage " + std::to_string(i++) + ": " + stage.print() + "\n";
135  }
136  i = 0;
137  double coef = 0.5;
138  double min = 0.0;
139  double max = 0.0;
140  for (auto& stage : stages) {
141  text += "group delay at order " + std::to_string(i + 1) + ": ";
142  double stageMin = coef * stage.getMinGroupDelay();
143  double stageMax = coef * stage.getMaxGroupDelay();
144  min += stageMin;
145  max += stageMax;
146  text += "min = " + std::to_string(min) + ", ";
147  text += "max = " + std::to_string(max) + ". ";
148  text += "(stage " + std::to_string(i++) + " group delay: ";
149  text += "min = " + std::to_string(stageMin) + ", ";
150  text += "max = " + std::to_string(stageMax);
151  text += ")\n";
152  coef *= 0.5;
153  }
154  double baseTransition = stages[0].transition;
155  text += "linear badwith at 44.1 KHz = " +
156  std::to_string(22050.0 - baseTransition * 44100.0) + " Hz\n";
157  text += "linear badwith at 96 KHz = " +
158  std::to_string(48000.0 - baseTransition * 96000.0) + " Hz\n";
159  text += "linear badwith at 192 KHz = " +
160  std::to_string(96000.0 - baseTransition * 192000.0) + " Hz\n";
161  text += "\n";
162  return text;
163 }

273 inline void
274 IirOversamplingDesigner::GroupDelayGraph::fromStages(
275  std::vector<Stage> const& stages,
276  int resolution)
277 {
278  std::vector<double> coefs;
279  graph.resize(resolution, 0.0);
280  double f = 0.5 / resolution;
281  double coef = 0.5;
282  for (auto& stage : stages) {
283  for (int i = 0; i < resolution; ++i) {
284  graph[i] += coef * stage.getGroupDelay(i * f);
285  }
286  coef *= 0.5;
287  }
288 }
