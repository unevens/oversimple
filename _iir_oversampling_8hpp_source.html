711  void ProcessBlock(InterleavedBuffer<Scalar> const& input,
712  int numSamples,
713  int numChannelsToProcess) override
714  {
715  if (numChannelsToProcess < 0) {
716  numChannelsToProcess = this->numChannels;
717  }
718  assert(numChannelsToProcess <= this->numChannels);
719  assert(numChannelsToProcess <= input.GetNumChannels());
720  this->PrepareBuffer(numSamples);
721  output = &this->buffer[0];
722  auto& temp = this->buffer[1];
723 
724  switch (this->order) {
725  case 0: {
726  output = const_cast<InterleavedBuffer<Scalar>*>(&input);
727  } break;
728  case 1: {
729  this->ApplyStage0(*output, input, numSamples / 2, numChannelsToProcess);
730  } break;
731  case 2: {
732  this->ApplyStage1(temp, input, numSamples / 2, numChannelsToProcess);
733  this->ApplyStage0(*output, temp, numSamples / 4, numChannelsToProcess);
734  } break;
735  case 3: {
736  this->ApplyStage2(*output, input, numSamples / 2, numChannelsToProcess);
737  this->ApplyStage1(temp, *output, numSamples / 4, numChannelsToProcess);
738  this->ApplyStage0(*output, temp, numSamples / 8, numChannelsToProcess);
739  } break;
740  case 4: {
741  this->ApplyStage3(temp, input, numSamples / 2, numChannelsToProcess);
742  this->ApplyStage2(*output, temp, numSamples / 4, numChannelsToProcess);
743  this->ApplyStage1(temp, *output, numSamples / 8, numChannelsToProcess);
744  this->ApplyStage0(*output, temp, numSamples / 16, numChannelsToProcess);
745  } break;
746  case 5: {
747  this->ApplyStage4(*output, input, numSamples / 2, numChannelsToProcess);
748  this->ApplyStage3(temp, *output, numSamples / 4, numChannelsToProcess);
749  this->ApplyStage2(*output, temp, numSamples / 8, numChannelsToProcess);
750  this->ApplyStage1(temp, *output, numSamples / 16, numChannelsToProcess);
751  this->ApplyStage0(*output, temp, numSamples / 32, numChannelsToProcess);
752  } break;
753  default:
754  assert(false);
755  }
756  }
757 
758  InterleavedBuffer<Scalar>& GetOutput() override
759  {
760  assert(output);
761  return *output;
762  }
